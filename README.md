if num > 40 and num % 7 == 0:                       #verifica si el numero es divisible por 7 y mayor a 40

else:                                                               #En caso de no cumplir el if
    print("El número", num, "no cumple con las condiciones.")       #Imprime si el numero no cumple las cosas
